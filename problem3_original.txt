SELECT c.cname, p.pname
FROM Customer c
JOIN Orders o       ON c.cid = o.cid
JOIN OrderItem oi   ON o.oid = oi.oid
JOIN Product p      ON p.pid = oi.pid
WHERE c.region = 'West'
  AND p.category LIKE 'Electronics%'
  AND o.amount > 200;
